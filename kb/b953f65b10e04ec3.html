function RandomBinom(n, p: Extended): Extended;
{ биноминальное распределение }
var
  x: Extended;
  i: Integer;
begin
  x := 0;
  for i := 1 to Round(n) do
    if Random - p <= 0 then x := x + 1;
  Result := x
end;

Пример использования:

x := RandomBinom(10, 0.4);
