function PADL(Src: string; Lg: Integer): string;
begin
  Result := Src;
  while Length(Result) < Lg do
    Result := ' ' + Result;
end;

function PADC(Src: string; Lg: Integer): string;
begin
  Result := Src;
  while Length(Result) < Lg do
  begin
    Result := Result + ' ';
    if Length(Result) < Lg then
    begin
      Result := ' ' + Result;
    end;
  end;
end;
