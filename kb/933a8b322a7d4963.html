unit RowGrid;
 
interface
 
uses
  WinTypes, WinProcs, Classes, Grids, DBGrids;
 
type
  TRowDBGrid = class(TDBGrid)
  public
    property Row;
    property RowCount;
    property VisibleRowCount;
  end;
 
procedure Register;
 
implementation
 
procedure Register;
begin
  RegisterComponents('Data Controls', [TRowDBGrid]);
end;
 
end.
