Как включить / отключить хранитель экрана?

01.01.2010
procedure TForm1.Button1Click(Sender: TObject);
begin
{Turn it off}
  SystemParametersInfo(SPI_SETSCREENSAVEACTIVE,
                       0,
                       nil,
                       0);
{Turn it on}
  SystemParametersInfo(SPI_SETSCREENSAVEACTIVE,
                       1,
                       nil,
                       0);
end;
