function RandomGipgeo(tn, ns, p: Extended): Extended;
{ гипергеометрическое распределение }
var
  x, s: Extended;
  i: Integer;
begin
  x := 0;
  for i := 1 to Round(ns) do begin
    if Random - p <= 0
      then begin
        s := 1;
        x := x + 1
      end
      else s := 0;
    p := (tn * p - s) / (tn - 1)
  end;
  Result := x
end;

Пример использования:

x := RandomGipgeo(2.5, 3, 0.4);
