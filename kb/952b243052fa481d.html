type
  TTypeStr = (tsString, tsDate, tsNumber);
 
function CheckString(const Value: string): TTypeStr;
begin
  if StrToDateTimeDef(Value, 0) = 0 then
    if StrToIntDef(Value, 0) = 0 then
      Result := tsString
    else
      Result := tsNumber
  else
    Result := tsDate
end;
