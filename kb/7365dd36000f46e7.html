Как показать удаленные записи

01.01.2010
procedure DeletedRecords(Table: TTable; SioNo: Boolean);
begin
  Table.DisableControls;
  try
    Check(DbiSetProp(hDBIObj(Table.Handle), curSOFTDELETEON, Longint(SioNo)));
  finally
    Table.EnableControls;
  end;
  Table.Refresh;
end;
