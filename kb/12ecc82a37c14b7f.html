function GetWindowsFolder:string;

 
var p:PChar;
begin
  GetMem(p, MAX_PATH);
  result:='';
  if GetWindowsDirectory(p, MAX_PATH)>0 then
    result:=string(p);
  FreeMem(p);
end;
